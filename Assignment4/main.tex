\section*{Problem}
A fair dice is rolled twice. The probability that an odd number will follow an even number is
\begin{enumerate}[label=(\alph*)]
\begin{multicols}{4}
\item $\dfrac{1}{2}$
\item $\dfrac{1}{6}$
\item $\dfrac{1}{3}$
\item $\dfrac{1}{4}$
\end{multicols}
\end{enumerate}

\maketitle
\section*{Solution}
Let a random variable $X \in \{0,1\}$ denote the possible outcomes of the event of rolling a die.
\begin{table}[h]
\centering
% \resizebox{\columnwidth}{!}{
    \begin{tabular}{|c|c|c|}
        \hline
        Dice output & X & Number\\
        \hline
        Even number & 0 & $n(X=0)=3$ \\
        \hline
        Odd number & 1 & $n(X=1)=3$\\
        \hline
        \end{tabular}
% }
\\
\caption{Outcome of the Experiment} 
\label{table=1}
\end{table}
\begin{align}
    \pr{X=0} &= \frac{n(X=0)}{n(X=0)+n(X=1)} = \frac{3}{6} \\
    &= \frac{1}{2}\\
    \pr{X=1} &= \frac{n(X=1)}{n(X=0)+n(X=1)} = \frac{3}{6} \\
    &= \frac{1}{2}
\end{align}
If $E$ denotes the event that an odd number follows an even then,
\begin{align}
    \pr{E} &= \pr{X=0} \times \pr{X=1}\\
    &= \frac{1}{2}\times\frac{1}{2}\\
    &= \frac{1}{4}
\end{align}
\\The required probability is $\dfrac{1}{4}$; option (d) is correct.
